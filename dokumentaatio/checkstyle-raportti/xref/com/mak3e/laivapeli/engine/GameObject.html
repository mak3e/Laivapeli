6   package com.mak3e.laivapeli.engine;
7   
8   import java.awt.Graphics2D;
9   import java.awt.Image;
10  import java.awt.Point;
11  import java.awt.image.BufferedImage;
12  import java.io.File;
13  import java.io.IOException;
14  import javax.imageio.ImageIO;
15  
16  /**
17   *
18   * @author Make
19   */
20  public class GameObject {
21  
22      //1 unit is about 10m
23      //1 unit is 256 pixels when width is 800 and camera zoom is 1
24      //and sprites are rendered 4 times smaller
25      private Vector2 pos;
26      private int layer = 0;
27      private float angle = 0f; //Angle
28      private Image sprite; //Sprite
29  
30      public GameObject(Vector2 pos, int layer) {
31          this.pos = pos;
32          this.layer = layer;
33      }
34  
35      public GameObject(Vector2 pos, int layer, String sprite) {
36          this(pos, layer);
37          this.sprite = load(sprite);
38      }
39  
40      public void capture(Camera camera) {
41          if (sprite != null && camera.getLayer() >= this.layer) {
42              camera.drawSprite(pos, angle, sprite);
43          }
44      }
45  
46      public void update() {
47          // Subclasses will use this method for updating
48      }
49  
50      public Vector2 getPos() {
51          return pos;
52      }
53  
54      public int getLayer() {
55          return layer;
56      }
57  
58      public float getA() {
59          return angle;
60      }
61      
62      BufferedImage load(String path) {
63          BufferedImage image = null;
64          try {                
65              image = ImageIO.read(new File("resources" + File.separator + path));
66          } catch (IOException ex) {
67              System.out.println("Error while loading a resource");
68          }
69          return image;
70          
71         
72      }
73  }
