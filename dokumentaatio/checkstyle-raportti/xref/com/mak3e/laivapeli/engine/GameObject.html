6   package com.mak3e.laivapeli.engine;
7   
8   import java.awt.Graphics2D;
9   import java.awt.Image;
10  import java.awt.Point;
11  
12  /**
13   *
14   * @author Make
15   */
16  public class GameObject {
17  
18      //1 unit is about 10m
19      //1 unit is 256 pixels when width is 800 and camera zoom is 1
20      //and sprites are rendered 4 times smaller
21      private Vector2 pos;
22      private int layer = 0;
23      private float angle = 0f; //Angle
24      private Image sprite; //Sprite
25  
26      public GameObject(Vector2 pos, int layer) {
27          this.pos = pos;
28          this.layer = layer;
29      }
30  
31      public GameObject(Vector2 pos, int layer, Image sprite) {
32          this(pos, layer);
33          this.sprite = sprite;
34      }
35  
36      public void capture(Camera camera) {
37          if (sprite != null && camera.getLayer() >= this.layer) {
38              camera.drawSprite(pos, angle, sprite);
39          }
40      }
41  
42      public void update() {
43          // Subclasses will use this method for updating
44      }
45  
46      public Vector2 getPos() {
47          return pos;
48      }
49  
50      public int getLayer() {
51          return layer;
52      }
53  
54      public float getA() {
55          return angle;
56      }
57  }
