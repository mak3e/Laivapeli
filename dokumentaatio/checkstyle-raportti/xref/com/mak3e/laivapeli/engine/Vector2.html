6   package com.mak3e.laivapeli.engine;
7   
8   /**
9    *
10   * @author Make
11   */
12  public class Vector2 {
13  
14      public final float x;
15      public final float y;
16  
17      public Vector2(float x, float y) {
18          this.x = x;
19          this.y = y;
20      }
21  
22      public Vector2(int x, int y) {
23          this.x = (float) x;
24          this.y = (float) y;
25      }
26  
27      public Vector2 add(Vector2 pos) {
28          return new Vector2(this.x + pos.x, this.y + pos.y);
29      }
30  
31      public Vector2 subtract(Vector2 pos) {
32          return new Vector2(this.x - pos.x, this.y - pos.y);
33      }
34  
35      public boolean isEqualTo(Vector2 pos) {
36          return x == pos.x && y == pos.y;
37      }
38  
39      @Override
40      public String toString() {
41          return "(" + x + ", " + y + ")";
42      }
43  
44  }
