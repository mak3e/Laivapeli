6   package com.mak3e.laivapeli.game;
7   
8   import com.mak3e.laivapeli.engine.Camera;
9   import com.mak3e.laivapeli.engine.Core;
10  import com.mak3e.laivapeli.engine.GameObject;
11  import com.mak3e.laivapeli.engine.Vector2;
12  
13  /**
14   * Water, covers 70% of earths surface.
15   *
16   * @author Make
17   */
18  public class Water extends GameObject {
19  
20      private final int springsPerUnit = 10;
21      private Spring[] springs;
22      private float width;
23      private final float tension = 0.025f;
24      private final float dampening = 0.025f;
25      private final float spread = 0.1f;
26      private final int smoothing = 8;
27      private float stepTime = 0f;
28      private float stepsPerSecond = 30f;
29      private float stepDelta = 1f / stepsPerSecond;
30  
31      /**
32       * Constructor.
33       *
34       * @param pos position
35       * @param levelWidth level width in units
36       */
37      public Water(Vector2 pos, float levelWidth) {
38          super(pos);
39          this.width = levelWidth;
40          springs = new Spring[(int) (this.width * springsPerUnit)];
41          for (int i = 0; i < springs.length; i++) { // OPTIMIZE THIS
42              springs[i] = new Spring();
43          }
44          springs[1].height = -10f;
45      }
46  
47      /**
48       * Get height of the water at x position.
49       *
50       * @param x x position
51       * @return height as float
52       */
53      public float getHeight(float x) {
54          Vector2 height = getClosestSpringHeights(x);
55          return heightBetween(height.x, height.y, x - (float) Math.floor(x));
56      }
57  
58      /**
59       * Get angle of the water at x position.
60       *
61       * @param x x position
62       * @return angle in degrees as float
63       */
64      public float getAngle(float x) {
65          Vector2 height = getClosestSpringHeights(x);
66          return angleOf(height.x - height.y, 1f / springsPerUnit);
67      }
68  
69      Vector2 getClosestSpringHeights(float x) {
70          if (inRange(x)) {
71              x *= springsPerUnit;
72              int leftSpring = (int) Math.floor(x);
73              int rightSpring = (int) Math.ceil(x);
74              if (leftSpring == rightSpring) {
75                  if (leftSpring > 0) {
76                      leftSpring -= 1;
77                  } else {
78                      rightSpring += 1;
79                  }
80              }
81              float height1 = springs[leftSpring].height;
82              float height2 = springs[rightSpring].height;
83              return new Vector2(height1, height2);
84          }
85          return new Vector2(0, 0);
86      }
87  
88      boolean inRange(float x) {
89          return (x >= 0f && x <= width);
90      }
91  
92      float angleOf(float height, float width) {
93          return (float) Math.toDegrees(Math.atan(height / width));
94      }
95  
96      float heightBetween(float height1, float height2, float x) {
97          return (height1 * x + height2 * (1 - x));
98      }
99  
100     @Override
101     public void update() {
102         float deltaTime = Core.engine.getClock().getDeltaTime();
103         while (stepTime > stepDelta) {
104             for (int i = 0; i < springs.length; i++) {
105                 springs[i].step(dampening, tension);
106             }
107             for (int i = 0; i < smoothing; i++) {
108                 smoothWaves();
109             }
110             stepTime -= stepDelta;
111         }
112         for (int i = 0; i < springs.length; i++) {
113             springs[i].update(deltaTime * stepsPerSecond);
114         }
115         stepTime += deltaTime;
116     }
117 
118     void smoothWaves() {
119         float delta = 0;
120         float lastDelta = 0;
121         for (int i = 0; i < springs.length - 1; i++) {
122             delta = nextDelta(
123                     springs[i].height,
124                     springs[i + 1].height,
125                     lastDelta
126             );
127             springs[i].velocity += delta;
128             springs[i].height += delta;
129             lastDelta += delta;
130         }
131     }
132 
133     float nextDelta(float height1, float height2, float lastDelta) {
134         return spread * (height2 - height1) - lastDelta;
135     }
136 
137     @Override
138     public void capture(Camera camera) {
139         Vector2 pos = super.getPos();
140         Vector2[] points = new Vector2[springs.length + 2];
141         for (int i = 0; i < springs.length; i++) {
142             points[i] = new Vector2(
143                     pos.x + ((i / (float) (springs.length - 1)) * width),
144                     pos.y + springs[i].height
145             );
146         }
147         float bottom = -camera.getUnitsPerHeight() / 2 + camera.getPos().y - 1;
148         points[springs.length] = new Vector2(this.width, bottom);
149         points[springs.length + 1] = new Vector2(pos.x, bottom);
150         camera.drawPolygon(points);
151     }
152 }
153 
154 class Spring {
155 
156     public float height = 0f;
157     public float velocity = 0f;
158 
159     public void update(float deltaStep) {
160         height += velocity * deltaStep;
161     }
162 
163     public void step(float dampening, float tension) {
164         velocity += -height * tension - velocity * dampening;
165     }
166 
167 }
